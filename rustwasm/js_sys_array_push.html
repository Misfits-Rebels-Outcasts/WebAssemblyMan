js_sys::Array::push function

Example

This example adds one or more elements to the end of an array and returns the length of the array.


pub fn array_push() {
    let arr = js_sys::Array::new();
    arr.push(&JsValue::from("John"));
    web_sys::console::log_2(&"Array Length: %d".into(),
                            &JsValue::from(arr.push(&JsValue::from("Jane")))
                            );
    web_sys::console::log_2(&"Array Length: %d".into(),
                            &JsValue::from(arr.push(&JsValue::from("James")))
                            );
}
